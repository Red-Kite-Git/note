# 前端架构设计

```
project-root/
├── api/                    # API接口管理
│   └── index.js            # API入口文件
├── assets/                 # 静态资源
│   ├── images/             # 图片资源
│   ├── styles/             # 样式资源
│   └── fonts/              # 字体资源
├── components/             # 组件
│   ├── common/             # 通用组件
│   └── layout/             # 布局组件
├── config/                 # 配置文件
│   └── config.js           # 项目配置
├── router/                 # 路由
│   └── index.js            # 路由配置
├── store/                  # Vuex状态管理
│   ├── modules/            # 模块
│   └── index.js            # 入口
├── utils/                  # 工具函数
│   ├── request.js          # 请求封装
│   └── util.js             # 通用工具
└── views/                  # 视图组件
    ├── Home.vue            # 首页
    └── About.vue           # 关于页
```
